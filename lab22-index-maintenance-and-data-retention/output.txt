------------------------------------------------------------
Task 2.2: Adjust retention policy (keep 30 days)
------------------------------------------------------------

toor@ip-172-31-10-221:~$ sudo -u postgres psql -d labdb -c "DELETE FROM app_logs WHERE ts < now() - interval '30 days';"
DELETE 0

toor@ip-172-31-10-221:~$ sudo -u postgres psql -d labdb -c "CREATE OR REPLACE FUNCTION enforce_log_retention(days_to_keep INT) RETURNS VOID AS $$ BEGIN DELETE FROM app_logs WHERE ts < now() - (days_to_keep || ' days')::interval; END; $$ LANGUAGE plpgsql;"
CREATE FUNCTION
